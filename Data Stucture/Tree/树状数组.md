# 树状数组

```C++
struct FenwickTree {
    int lb(int x) { return x & -x; }
    void add(int x, int v) {
        for (; x < n; x += lb(x)) tr[x] += v;
    }
    int que(int x) {
        int sum = 0;
        for (; ~x; x -= lb(x)) sum += tr[x];
        return sum;
    }
    void clear(int v = 0) {
        for (auto &_: tr) _ = v;
    }
    FenwickTree(int _n, int v = 0): n(_n) {
        tr.resize(n, v);
    }
    int n;
    std::vector<int> tr;
};
```
