# 树状数组

```C++
template<typename T>
struct FenwickTree {
    int lb(int x) { return x & -x; }
    void add(int x, T v) {
        for (; x <= n; x += lb(x)) tr[x] += v;
    }
    T que(int x) {
        T sum = T(0);
        for (; x; x -= lb(x)) sum += tr[x];
        return sum;
    }
    void clear(T v = 0) {
        for (auto &_: tr) _ = v;
    }
    FenwickTree(int _n, T v = T(0)): n(_n) {
        tr.resize(n + 1, v);
    }
    int n;
    std::vector<T> tr;
};
```
