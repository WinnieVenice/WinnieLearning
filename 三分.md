const double eps = 1e-5;
double f(double x, double y) {

        double l = 0, r = 100; 
        double ans = 0;
        while (r - l > eps) {
            double lmid = (l + r) / 2, rmid = (lmid + r) / 2;
            if (f(lmid, x) < f(rmid, x)) {
                ans = lmid;
                r = rmid;
            } else {
                l = lmid;
